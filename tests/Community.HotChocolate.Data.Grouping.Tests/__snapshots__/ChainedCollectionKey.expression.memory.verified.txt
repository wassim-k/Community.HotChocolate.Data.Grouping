EnumerableQuery
    .SelectMany(
        src => src.Projects ?? Enumerable.Empty(),
        (src, e0) => new AnonymousType(src, e0))
    .SelectMany(
        c1 => c1.Item2.Tasks ?? Enumerable.Empty(),
        (c1, e1) => new AnonymousType(c1.Item1, c1.Item2, e1))
    .GroupBy(src => new AnonymousType(src.Item3.Name))
    .Select(g => new AnonymousType(g.Key, g.Count()))
